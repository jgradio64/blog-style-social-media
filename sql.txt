create table Users (
	UserID	SERIAL PRIMARY KEY,
	UserName VARCHAR(20) NOT NULL,
	About VARCHAR(200)
	);
	
	
	
create table Posts (
	PostID SERIAL PRIMARY KEY,
	UserID INT,
	Post VARCHAR(400),
	CONSTRAINT fk_Users
		FOREIGN KEY(UserID)
			REFERENCES Users(UsersID)
	);
